void Inverser(Stack &S, Queue &Q) {
	while(!QueueEmpty(Q)) {
		x = DeQueue(Q);
		Push(S, x);
	}
	while(!StackEmpty(S)) {
		Pop(S, x);
		EnQueue(Q, x);
	}
}

栈的存储结构#

顺序#

定义（这里令top指向栈顶元素）

#define MaxSize 10
typedef struct {
	ElemType data[MaxSize];
	int top;
} SqStack;

初始化

SqStack S;
InitStack(S); //令 top=-1

进栈

bool Push(SqStack &S, ElemType x){
	if (S.top == MaxSize-1)
		return false;
	S.top = S.top + 1;
	S.data[S.top] = x;
	// 上面两句也可以写成 S.data[++S.top] = x;
	return true;
}

出栈（数据还残留在内存中，只是逻辑上删除）

bool Pop(SqStack &S, ElemType &x){
	if (S.top == -1)
		return false;
	x = S.data[S.top];
	S.top = S.top - 1;
	// 上面两句也可以写成 x = S.data[S.top--];
	return true;
}

==共享栈==：两个顺序栈共用存储空间，分别从两边往中间靠拢

链式#

相当于只在==头==进行插入删除的单链表

  • 插入：头插法建立单链表
  • 删除：头结点“后删”操作

两种实现方式

  • 带头结点

  • 不带头结点（推荐）

    • 实现

      typedef int ElemType;

typedef struct Linknode {
	ElemType data;
	struct Linknode *next;
}Linknode, *LiStack;

void InitStack(LiStack &S) {
    S = NULL;
}

void Push(LiStack &S, ElemType x) {
    Linknode* node = (Linknode*)malloc(sizeof(Linknode));
    node->data = x;
    if (S==NULL) {
        S = node;
        node->next = NULL;
    }
    else {
        node->next = S;
        S = node;
    }
}

bool Pop(LiStack &S, ElemType &x) {
    if (S==NULL)
        return false;
    x = S->data;
    Linknode* tmp = S;
    S = S->next;
    free(tmp);
    return true;
}

bool GetTop(LiStack &S, ElemType &x) {
    if (S==NULL)
        return false;
    x = S->data;
    return true;
}

bool StackEmpty(LiStack S) {
    if (S==NULL)
        return true;
    else
        return false;
}

栈的应用#

（还包括进制转换、迷宫求解等）

括号匹配#

依次扫描所有字符，遇到左括号入栈，遇到右括号则弹出栈顶元素检查是否匹配。

匹配成功情况：括号序列处理结束时，栈空

匹配失败情况：

  • 左括号单身：栈非空，括号序列空
  • 右括号单身：栈空，括号序列非空
  • 左右括号不匹配：匹配失败

表达式求值#

  • 前缀表达式（波兰表达式）

  • ==后缀表达式==（逆波兰表达式）

  • 中缀表达式

==中缀转后缀==（转前缀同理）

  • 手算方法：先确定各个运算符的运算顺序（左优先原则，保证结果唯一，手算和机算结果相同，确保算法的确定性），然后按照 ⌈左操作数 右操作数 运算符⌋ 的方式组合。

  • 机算方法：用栈
    • 遇到操作数。直接加入后缀表达式
    • 遇到界限符。遇到(直接入栈；遇到)，则依次弹出栈内(之前的所有运算符并加入后缀表达式
    • 遇到运算符。依次弹出栈内优先级高于或等于它的所有运算符，并加入后缀表达式，若碰到(或栈空则停止。之后再把当前运算符入栈。

==后缀表达式求值==（前缀同理）

  • 手算方法：从左到右扫描，每遇到一个运算符，就让运算符前面最近的两个操作数执行对应运算，合体为一个操作数。

  • 机算方法：用栈

    依次扫描每个元素入栈，如果扫描到运算符，则弹出两个栈顶元素（先出栈的是右操作数），执行相应运算，将结果压回栈顶，直到处理完所有元素。

==中缀表达式的计算（上面两个的结合）==

用到两个栈：

  • 操作数栈：存放暂时不能确定运算顺序的操作数

  • 运算符栈：存放暂时不能确定运算顺序的运算符

过程：

  • 扫描到操作数：压入操作数栈
  • 扫描到运算符或界限符：按照“中缀转后缀”相同的逻辑压入运算符栈，期间每弹出一个运算符时，就从操作数栈中弹出两个执行相应运算，再将结果压回操作数栈

栈在递归中的应用#

函数调用特点：最后被调用的函数最先执行结束（LIFO）。

函数调用时，用“函数调用栈”（递归工作栈）存储：

  • 调用返回地址
  • 实参
  • 局部变量

每进入一层递归，就将递归调用所需信息压入栈顶；每推出一层循环，就从栈顶弹出相应信息。

缺点：效率低，太多层递归可能导致栈溢出；可能包含很多重复机算。

消除递归，即采用非递归方式重写递归程序时必须使用栈：错，比如求斐波那契数列只需一个循环即可

队列的存储结构#

顺序#

定义（这里令rear指向下一个要插入的位置）

#define MaxSize 10
typedef struct {
	ElemType data[MaxSize];
	int front, rear;
} SqQueue;

初始化

front=0, rear=0

判空条件

rear == front

入队（只能从队尾插入）

if ((Q.rear+1)%MaxSize==Q.front) // 队列满
	return false;
Q.data[Q.rear] = x;
Q.rear = (Q.rear + 1) % MaxSize;
return true;

出队（只能从对头删除）

if (Q.rear==Q.front) // 队列空
	return false;
x = Q.data[Q.front];
Q.front = (Q.front + 1) % MaxSize;
return true;

==入队如何判断队列已满？==

rear==MaxSize NO

(Q.rear+1)%MaxSize==Q.front Q.rear = (Q.rear + 1) % MaxSize; YES，用模运算将存储空间在逻辑上变成环状（循环队列）

若不牺牲一个存储单元，则继续入队，rear=front，会影响判空条件。

队列元素个数：(rear+MaxSize-front)%MaxSize

==如何不浪费空间？==

  1. 设置变量size记录队列长度

    初始时size=0，队满条件 size==MaxSize

  2. 设置标记变量tag

    每次删除操作成功时令tag=0，插入操作成功时令tag=1

    队满条件 front==rear && tag==1

    队空条件 rear==front && tag==0

    进队 Q.data[Q.rear] = x; Q.rear = (Q.rear +1)%MaxSize; Q.tag=1;

    出队 x = Q.data[Q.front]; Q.front=(Q.front+1)%MaxSize; Q.tag=0;

链式#

用单链表实现队列时，通常在==链表尾==进行插入在==链表头==进行删除，因为表头便于删除元素（表尾删除元素需要找到前驱节点），因此可以使用

  • 带头尾指针的单链表（不用带头尾指针的循环链表，因为进出队列后还要修改成循环的，麻烦）

  • 带尾指针的循环单链表

  • 只带头指针的非循环双链表（错）插入O(n)

两种实现方式

  • 带头结点
  • 不带头结点

定义

typedef struct LinkNode { // 链式队列结点
	ElemType data;
	struct LinkNode *next;
}LinkNode;

typedef struct { // 链式队列
	LinkNode *front, *rear; // 队列的队头和队尾指针
}LinkQueue;

初始化（带头结点）

void InitQueue(LinkQueue &Q) {
	Q.front = Q.rear = (LinkNode*)malloc(sizeof(LinkNode));
	Q.front->next = NULL;
}

判空（带头结点）

Q.front==Q.rear
或 Q.front->next==NULL

初始化（不带头结点）

void InitQueue(LinkQueue &Q) {
	Q.front = NULL;
	Q.rear = NULL;
}

判空（不带头结点）

Q.front==NULL
或 Q.rear==NULL
